// api/search.js export default async function handler(req, res) { if (req.method !== 'POST') return res.status(405).json({ error: 'Method not allowed' }); const { query } = req.body; const GROQ_API_KEY = process.env.GROQ_API_KEY; // Ensure this matches your Vercel variable name try { const response = await fetch('https://api.groq.com/openai/v1/chat/completions', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${GROQ_API_KEY}` }, body: JSON.stringify({ model: 'llama3-70b-8192', // Or 'llama3-8b-8192' for even more speed messages: [{ role: 'user', content: `You are a shopping AI. User searched: "${query}". Return exactly 4 product results as JSON array with this format: [{"name":"Product Name","description":"2 sentence description","rating":"4.5/5","price":"$XX - $XX","status":"safe OR caution OR scam","reason":"one line reason for status"}] Return ONLY the JSON array.` }], response_format: { type: "json_object" } // Groq handles JSON very efficiently }) }); const data = await response.json(); res.status(200).json(data); } catch (error) { res.status(500).json({ error: 'Search failed' }); } }
